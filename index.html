#!/usr/bin/env python3 """ process_video.py Minimal cricket ball detector + speed estimator prototype. Usage: python process_video.py --video input.mp4 --output output.mp4 --model models/yolov8-ball.pt One-time calibration: click 4 image points (clockwise or any order) on the pitch that map to known real-world coordinates (enter the x,y in meters for each clicked point). The script computes a homography to map image->ground plane (meters). Notes: - Requires a YOLOv8 model trained to detect the ball (single class). - This prototype assumes one dominant ball detection per frame. """ import argparse import cv2 import numpy as np import time import pandas as pd from ultralytics import YOLO from scipy.spatial.distance import euclidean from tqdm import tqdm # Speed classification thresholds (km/h) # You can tune these thresholds for your context FAST_KPH = 140 MEDIUM_KPH = 120 def parse_args(): p = argparse.ArgumentParser() p.add_argument("--video", required=True, help="Input video file") p.add_argument("--output", default="output.mp4", help="Annotated output video") p.add_argument("--model", default="models/yolov8-ball.pt", help="YOLOv8 model path") p.add_argument("--conf", type=float, default=0.25, help="Detection confidence threshold") p.add_argument("--min_area", type=int, default=20, help="Minimum bbox area in pixels to keep") p.add_argument("--smooth_n", type=int, default=5, help="Smoothing window for speed (frames)") return p.parse_args() def click_points_and_get_homography(frame): """ Interactive collection of 4 image points and mapping to real-world meters. Returns homography H (3x3) mapping image (px) to ground plane (meters). """ pts = [] clone = frame.copy() window = "Click 4 ground points (corners/markers). Press ESC to cancel." cv2.namedWindow(window) def on_mouse(event, x, y, flags, param): if event == cv2.EVENT_LBUTTONDOWN: pts.append((x, y)) cv2.circle(clone, (x, y), 5, (0, 255, 0), -1) cv2.imshow(window, clone) cv2.setMouseCallback(window, on_mouse) cv2.imshow(window, clone) print("Click 4 image points on the pitch (clockwise or any order).") while True: cv2.imshow(window, clone) key = cv2.waitKey(1) & 0xFF if key == 27: # ESC break if len(pts) >= 4: break cv2.destroyWindow(window) if len(pts) < 4: raise RuntimeError("Calibration cancelled or not enough points.") pts = np.array(pts, dtype=np.float32) print("Enter the corresponding real-world coordinates (in meters) for the 4 clicked points.") world_pts = [] for i in range(4): s = input(f"Point {i+1} (format: x y) in meters: ") xw, yw = map(float, s.strip().split()) world_pts.append([xw, yw]) world_pts = np.array(world_pts, dtype=np.float32) H, status = cv2.findHomography(pts, world_pts, method=0) if H is None: raise RuntimeError("Homography estimation failed.") print("Homography computed.") return H def image_to_world(H, px): """ Map pixel (x,y) to world coordinates (meters) by homography H. px: (x, y) Returns (X, Y) in meters (2D ground). """ p = np.array([px[0], px[1], 1.0], dtype=np.float64) w = H @ p if w[2] == 0: return None w /= w[2] return (float(w[0]), float(w[1])) def classify_speed_kph(kph): if np.isnan(kph): return "unknown" if kph >= FAST_KPH: return "fast" if kph >= MEDIUM_KPH: return "medium" return "slow" def main(): args = parse_args() vid = cv2.VideoCapture(args.video) if not vid.isOpened(): raise RuntimeError("Cannot open video: " + args.video) fps = vid.get(cv2.CAP_PROP_FPS) or 30.0 width = int(vid.get(cv2.CAP_PROP_FRAME_WIDTH)) height = int(vid.get(cv2.CAP_PROP_FRAME_HEIGHT)) total_frames = int(vid.get(cv2.CAP_PROP_FRAME_COUNT) or 0) # Load YOLOv8 model print("Loading YOLOv8 model:", args.model) model = YOLO(args.model) # Calibration (homography) print("Extracting a frame for calibration...") ret, frame0 = vid.read() if not ret: raise RuntimeError("Failed to read frame for calibration") H = None try: H = click_points_and_get_homography(frame0) except Exception as e: print("Calibration failed:", e) print("Exiting.") return # Reset video to start vid.set(cv2.CAP_PROP_POS_FRAMES, 0) # Output writer fourcc = cv2.VideoWriter_fourcc(*"mp4v") out_vid = cv2.VideoWriter(args.output, fourcc, fps, (width, height)) # Process frames prev_world = None prev_time = None speeds = [] # history of instantaneous speeds (m/s) rows = [] frame_idx = 0 pbar = tqdm(total=total_frames or None, desc="Processing") while True: ret, frame = vid.read() if not ret: break t_now = frame_idx / fps # Run detection (single inference per frame) results = model.predict(frame, conf=args.conf, imgsz=640, verbose=False) # results is a list, we take first detections = [] if len(results) > 0: res = results[0] # res.boxes.xyxyn or .xyxy, .cls, .conf if hasattr(res, "boxes"): boxes = res.boxes for box in boxes: xyxy = box.xyxy.cpu().numpy().flatten() conf = float(box.conf.cpu().numpy().flatten()) if getattr(box, "conf", None) is not None else 0.0 cls = int(box.cls.cpu().numpy().flatten()) if getattr(box, "cls", None) is not None else 0 x1, y1, x2, y2 = map(int, xyxy[:4]) area = (x2 - x1) * (y2 - y1) if area < args.min_area: continue cx = int((x1 + x2) / 2) cy = int((y1 + y2) / 2) detections.append({"xyxy": (x1, y1, x2, y2), "conf": conf, "cls": cls, "centroid": (cx, cy)}) # Pick best detection (highest conf) if multiple ball_det = None if len(detections) > 0: detections = sorted(detections, key=lambda d: -d["conf"]) ball_det = detections[0] world_pt = None speed_mps = float("nan") kph = float("nan") style = "unknown" if ball_det is not None: cx, cy = ball_det["centroid"] # Map to world meters mapped = image_to_world(H, (cx, cy)) if mapped is not None: world_pt = mapped # (X, Y) meters if prev_world is not None: dt = t_now - prev_time if prev_time is not None else 1.0 / fps if dt <= 0: dt = 1.0 / fps dist = euclidean(world_pt, prev_world) # meters in ground plane speed_mps = dist / dt kph = speed_mps * 3.6 prev_world = world_pt prev_time = t_now # smoothing speeds.append(speed_mps if not np.isnan(speed_mps) else 0.0) if len(speeds) > args.smooth_n: smoothed = float(np.mean([s for s in speeds[-args.smooth_n:] if s > 0])) else: smoothed = float(np.mean([s for s in speeds if s > 0])) if any(np.array(speeds) > 0) else float("nan") if not np.isnan(smoothed) and smoothed > 0: kph = smoothed * 3.6 speed_mps = smoothed style = classify_speed_kph(kph) else: style = "unknown" # Annotate frame annotated = frame.copy() if ball_det is not None: x1, y1, x2, y2 = ball_det["xyxy"] cv2.rectangle(annotated, (x1, y1), (x2, y2), (0, 255, 0), 2) cx, cy = ball_det["centroid"] cv2.circle(annotated, (cx, cy), 4, (0, 255, 0), -1) # Overlay text text1 = f"Frame: {frame_idx}" text2 = f"Speed: {kph:.1f} km/h" if not np.isnan(kph) else "Speed: --" text3 = f"Style: {style}" cv2.putText(annotated, text1, (10, 20), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255, 255, 255), 2) cv2.putText(annotated, text2, (10, 45), cv2.FONT_HERSHEY_SIMPLEX, 0.8, (0, 255, 255), 2) cv2.putText(annotated, text3, (10, 75), cv2.FONT_HERSHEY_SIMPLEX, 0.8, (0, 200, 255), 2) out_vid.write(annotated) rows.append({ "frame": frame_idx, "time_s": t_now, "pixel_cx": ball_det["centroid"][0] if ball_det else None, "pixel_cy": ball_det["centroid"][1] if ball_det else None, "world_x_m": world_pt[0] if world_pt else None, "world_y_m": world_pt[1] if world_pt else None, "speed_mps": speed_mps if not np.isnan(speed_mps) else None, "speed_kph": kph if not np.isnan(kph) else None, "style": style }) frame_idx += 1 pbar.update(1) pbar.close() vid.release() out_vid.release() # Save CSV df = pd.DataFrame(rows) csv_out = args.output.replace(".mp4", ".csv") df.to_csv(csv_out, index=False) print("Processing complete. Annotated video:", args.output) print("Per-frame CSV:", csv_out) if __name__ == "__main__": main()
